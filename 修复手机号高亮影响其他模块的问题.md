# 修复手机号高亮影响其他模块的问题

## 问题描述

在内容查找、智能搜索、智能问答等非正则查询功能中，所有的高亮都错误地高亮了手机号。

**现象**：
- 内容查找：搜索结果中的手机号被高亮 ❌
- 智能搜索：搜索结果中的手机号被高亮 ❌
- 智能问答：回答内容中的手机号被高亮 ❌
- 正则查询：手机号应该被高亮 ✅

## 问题原因

在 `highlightAndEscape` 和 `highlightAndEscapeExact` 函数中添加了无条件的手机号高亮：

```javascript
// ❌ 错误：所有调用都会高亮手机号
function highlightAndEscape(text, keywords) {
    // 高亮关键词
    keywords.forEach(...)
    
    // 高亮手机号（11 位，1 开头，第二位 3-9）
    const phoneRegex = /1[3-9]\d{9}/g;
    result = result.replace(phoneRegex, '___HIGHLIGHT_START___$&___HIGHLIGHT_END___');
    
    // ...
}
```

这导致**所有**使用这两个函数的地方都会高亮手机号，而不仅仅是正则查询。

## 解决方案

### 1. 移除通用高亮函数中的手机号高亮

修改 `highlightAndEscape` 和 `highlightAndEscapeExact`，只高亮关键词：

```javascript
// ✅ 正确：只高亮关键词
function highlightAndEscape(text, keywords) {
    if (!keywords || keywords.length === 0) return escapeHtml(text);
    
    // 分词处理
    let finalKeywords = [];
    // ...
    
    // 只高亮关键词
    finalKeywords.forEach(keyword => {
        const regex = new RegExp(`(${escaped})`, 'gi');
        result = result.replace(regex, '___HIGHLIGHT_START___$1___HIGHLIGHT_END___');
    });
    
    // 不再高亮手机号
    result = escapeHtml(result);
    // ...
}
```

### 2. 创建专门用于正则查询的高亮函数

新增 `highlightAndEscapeWithPattern` 函数，支持 pattern 参数：

```javascript
// 专门用于正则查询的高亮函数
function highlightAndEscapeWithPattern(text, keywords, pattern) {
    if (!keywords && !pattern) return escapeHtml(text);
    
    let result = text;
    
    // 高亮关键词
    if (keywords && keywords.length > 0) {
        keywords.forEach(keyword => {
            const regex = new RegExp(`(${escaped})`, 'gi');
            result = result.replace(regex, '___HIGHLIGHT_START___$1___HIGHLIGHT_END___');
        });
    }
    
    // 只在有 pattern 且是手机号正则时高亮手机号
    if (pattern && (pattern.includes('1[3-9]') || pattern.includes('1[3456789]') || pattern.includes('\\d{9}'))) {
        const phoneRegex = /1[3-9]\d{9}/g;
        result = result.replace(phoneRegex, '___HIGHLIGHT_START___$&___HIGHLIGHT_END___');
    }
    
    result = escapeHtml(result);
    // ...
}
```

### 3. 修改 viewFile 函数

让 `viewFile` 函数根据是否有 pattern 选择不同的高亮策略：

```javascript
// 如果有 pattern，使用正则高亮；否则使用关键词高亮
if (apiPattern) {
    window.regexPattern = apiPattern;
    const highlightedContent = highlightRegexContent(content, apiPattern);
    const smartFormatted = formatSmartContent(content, [], false, apiPattern);
    const sectionFormatted = formatSectionContent(content, [], false, apiPattern);
    
    window.formattedContent = {
        raw: `<pre>${escapeHtml(content)}</pre>`,
        smart: smartFormatted,
        section: sectionFormatted
    };
    
    document.getElementById('modal-content').innerHTML = toolbar + smartFormatted;
} else {
    window.contentKeywords = keywords && keywords.length > 0 ? keywords : apiKeywords;
    window.exactMatch = exactMatch;
    
    const highlightFunc = exactMatch ? highlightAndEscapeExact : highlightAndEscape;
    document.getElementById('modal-content').innerHTML = toolbar + 
        '<div id="content-display" class="file-content">' + 
        highlightFunc(content, window.contentKeywords) + 
        '</div>';
}
```

### 4. 修改 setFormat 函数

让格式切换时也根据是否有 pattern 选择高亮函数：

```javascript
function setFormat(format) {
    const display = document.getElementById('content-display');
    const content = window.rawContent || '';
    const keywords = window.contentKeywords || [];
    const pattern = window.regexPattern || '';
    
    if (format === 'raw') {
        display.className = 'file-content';
        // 如果有 pattern，使用正则高亮；否则使用关键词高亮
        if (pattern) {
            display.innerHTML = highlightRegexContent(content, pattern);
        } else {
            const highlightFunc = window.exactMatch ? highlightAndEscapeExact : highlightAndEscape;
            display.innerHTML = highlightFunc(content, keywords);
        }
    } else if (format === 'smart') {
        display.className = 'formatted-content';
        display.innerHTML = formatSmartContent(content, keywords, window.exactMatch, pattern);
    } else if (format === 'section') {
        display.className = 'formatted-content';
        display.innerHTML = formatSectionContent(content, keywords, window.exactMatch, pattern);
    }
}
```

| 场景 | 使用的函数 | 高亮内容 |
|-----|-----------|---------|
| 内容查找 | `highlightAndEscape` | 只高亮关键词 |
| 智能搜索 | `highlightAndEscapeExact` | 只高亮关键词 |
| 智能问答 | `highlightAndEscape` | 只高亮关键词 |
| 正则查询（查看全文） | `highlightRegexContent` | 高亮正则匹配内容 |
| 正则查询（手机号） | `highlightRegexContent` | 高亮手机号 |
| 正则查询（非手机号） | `highlightRegexContent` | 不高亮手机号 |
